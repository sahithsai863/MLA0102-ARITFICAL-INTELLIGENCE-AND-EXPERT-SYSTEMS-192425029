FUNCTION DFS_CAVE_EXPLORATION(CAVE_MAP, CURRENT_JUNCTION, VISITED)

    IF VISITED is EMPTY THEN
        CREATE an empty VISITED list
    END IF

    ADD CURRENT_JUNCTION to VISITED

    PRINT "Drone enters :", CURRENT_JUNCTION

    FOR EACH NEXT_JUNCTION connected to CURRENT_JUNCTION DO

        IF NEXT_JUNCTION is NOT in VISITED THEN

            PRINT "Moving deeper from", CURRENT_JUNCTION, "to", NEXT_JUNCTION

            CALL DFS_CAVE_EXPLORATION(CAVE_MAP, NEXT_JUNCTION, VISITED)

            PRINT "Backtracking to", CURRENT_JUNCTION

        END IF

    END FOR

    RETURN VISITED

END FUNCTION


BEGIN MAIN PROGRAM

    CREATE CAVE_MAP

    SET START_JUNCTION = "Entrance"

    PRINT "Rescue Drone DFS Cave Exploration"

    EXPLORATION_ORDER ← DFS_CAVE_EXPLORATION(CAVE_MAP, START_JUNCTION)

    PRINT "Final Summary"

    PRINT EXPLORATION_ORDER

END MAIN PROGRAM

STOP
