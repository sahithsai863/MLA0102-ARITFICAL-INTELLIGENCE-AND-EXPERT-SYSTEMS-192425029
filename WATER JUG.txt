Initialize Visited Set

Insert (0,0) into Queue
Mark (0,0) as Visited

WHILE Queue is not empty DO

    Remove current state (A, B)

    IF A = Goal OR B = Goal THEN
        Print the sequence of states
        STOP
    END IF

    Generate the following next states:

        Fill Jug A
        Fill Jug B
        Empty Jug A

        Pour Jug A into Jug B

    FOR each generated state DO
        IF state is not visited THEN

        END IF
    END FOR

END WHILE
